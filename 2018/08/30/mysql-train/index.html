创建训练用的库表 source file

  • linux 上无法导入，请另存为utf8编码格式
# 创建数据库表
create database train_sql default charset utf8;
use train_sql;

#建学生信息表student
create table student
(
sno varchar(20) not null primary key,
sname varchar(20) not null,
ssex varchar(20) not null,
sbirthday datetime,
class varchar(20)

);
#建立教师表
create table teacher
(
tno varchar(20) not null primary key,
tname varchar(20) not null,
tsex varchar(20) not null,
tbirthday datetime,
prof varchar(20),
depart varchar(20) not null

);
#建立课程表course
create table course
(
cno varchar(20) not null primary key,
cname varchar(20) not null,
tno varchar(20) not null,
foreign key(tno) references teacher(tno)

);
#建立成绩表
create table score
(
sno varchar(20) not null,
foreign key(sno) references student(sno),
cno varchar(20) not null,
foreign key(cno) references course(cno),
degree decimal(5,3)

);

查询语句

  • 查询Student表中的所有记录的Sname、Ssex和Class列。

    select Sname, Ssex, Class from Student;

  • 查询教师所有的单位即不重复的Depart列 distinct

    select distinct Depart from teacher;

  • 查询Student表的所有记录。

    select * from student;

  • 查询Score表中成绩在60到80之间的所有记录 between

    select sname, degree from score left join student on student.sno = score.sno where degree between 60 and 80;

  • 查询Score表中成绩为85，86或88的记录 in

    select degree, sname from score left join student on student.sno = score.sno where degree in (85, 86, 88);

  • 查询Student表中“9503”班或性别为“女”的同学记录。

    # 使用 union 查询，速度快，代替 or进行查询

select sname, ssex, class from student where class=95031 union
select sname, ssex, class from student where ssex = '女';    

# 使用 or 查询，速度慢，不建议使用

select sname, ssex, class from student where class=95031 or ssex = '女'；

  • 以Class降序查询Student表的所有记录。

    select sno, sname, ssex, sbirthday, class from student order by class desc;

  • 以Cno升序、Degree降序查询Score表的所有记录。 

    #以多个字段进行排序，前面的字段排序完毕后，再对 后面的字段进行排序

select sno, cno, degree from score order by Cno , Degree desc;

  • 查询“95031”班的学生人数。

    select count(*) as class_people from student where class = 95031；

  • 查询Score表中的最高分的学生学号和课程号。（子查询或者排序）

    select sno, cno, degree from score order by degree desc limit 1;

  • 查询每门课的平均成绩。

    select cno, avg(degree) from score group by cno;

  • 查询Score表中至少有5名学生选修的并以3开头的课程的平均分数。

    select cno, avg(degree) from score where cno like '3%' group by cno having count(*) > 4;

  • 查询分数大于70，小于90的Sno列。

    select sno, degree from score where degree between 70 and 90;

  • 查询所有学生的Sname、Cno和Degree列。 关联查询

    select sname, cno, degree from student left join score on score.sno = student.sno ;

  • 查询所有学生的Sno、Cname和Degree列。

    select sno, cname, degree from score left join course on score.cno = course.cno ;

  • 查询所有学生的Sname、Cname和Degree列。

    select sname, cname, degree from student left join score on score.sno = student.sno left join course on course.cno = score.cno;

  • 查询“95033”班学生的平均分。

    select class, avg(degree) from score left join student on student.sno = score.sno group by class having class = 95033;

  • 假设使用如下命令建立了一个grade表：

  • 现查询所有同学的Sno、Cno和rank列。

    select Sno, cno, degree,rank from score a left join grade b on degree between b.low and b.upp;

  • 查询选修“3-105”课程的成绩高于“109”号同学成绩的所有同学的记录。

    select sname, degree from score left join student on student.sno = score.sno where cno = ‘3-105’ and degree > (select degree from score where sno = ‘109’ and cno=’3-105’);
